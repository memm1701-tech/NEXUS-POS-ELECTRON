async function iniciarSincronizacionSilenciosa() {

if (window.sincronizando) return;

// Encolar modificaciones locales al Servidor VPS Maestro

const cola = await api.obtenerColaSincronizacion();

if (cola.length > 0 && navigator.onLine) {

for (const registro of cola) {

const response = await fetch(`${SERVER_URL}/api/maestro/categorias/sincronizar`, {

method: 'POST',

body: JSON.stringify(registro.datos)

});

if (response.ok) await api.eliminarDeCola(registro.id);

}

}

}
